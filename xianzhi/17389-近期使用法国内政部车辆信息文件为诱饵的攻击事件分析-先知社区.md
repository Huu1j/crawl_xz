```
import base64
import gzip
from Crypto.Cipher import AES
from Crypto.Util.Padding import unpad
import io

def FXpA(graK):
    key = base64.b64decode("nTSLN+tYRgl3UojZwGQp1KzCUfudWAtKTha+YBcCV/Q=")
    iv = base64.b64decode("TDBvkVz5NOJG94+PvS7jmA==")
    cipher = AES.new(key, AES.MODE_CBC, iv)
    decrypted_data = cipher.decrypt(graK)
    return unpad(decrypted_data, AES.block_size)

def jipi(graK):
    with gzip.GzipFile(fileobj=io.BytesIO(graK), mode='rb') as gz:
        decompressed_data = gz.read()
    return decompressed_data

def main():
    sRpf = r'${SamplePath}\paris.bat'
    with open(sRpf, 'r') as file:
        ReSA = file.read().splitlines()
    pcWw = None
    for line in ReSA:
        if line.startswith('YStKW'):
            pcWw = line[5:]
            break
    if pcWw:
        Fkof = pcWw.split('\')
        utl = jipi(FXpA(base64.b64decode(Fkof[0].replace("#", "/").replace("@", "A"))))
        Egg = jipi(FXpA(base64.b64decode(Fkof[1].replace("#", "/").replace("@", "A"))))
        Kby = jipi(FXpA(base64.b64decode(Fkof[2].replace("#", "/").replace("@", "A"))))
        with open('utl_decrypted.bin', 'wb') as f:
            f.write(utl)
        with open('Egg_decrypted.bin', 'wb') as f:
            f.write(Egg)
        with open('Kby_decrypted.bin', 'wb') as f:
            f.write(Kby)

if __name__ == "__main__":
    main()
```

编写解密脚本如下

```
import struct
from ctypes import c_int32

def decrypt_string(data: bytes, ids: list[int]) -> list[str]:
    results = []
    for id in ids:
        if id >> 31:
            results.append("")
            continue

        base_offset = struct.unpack_from("<I", data, 0)[0] 
        offset = id ^ base_offset

        str_length = struct.unpack_from("<I", data, offset)[0]
        enc_bytes = bytearray(data[offset + 8:offset + 8 + str_length]) 

        key = struct.unpack_from("<I", data, offset + 4)[0] 
        key_bytes = [(key >> (8 * i)) & 0xFF for i in range(4)] 

        num = len(enc_bytes) - 1
        j = 0
        while j < num:
            enc_bytes[j] ^= enc_bytes[num]
            enc_bytes[num] ^= (enc_bytes[j] ^ key_bytes[j % 4])
            enc_bytes[j] ^= enc_bytes[num]
            j += 1
            num -= 1

        if len(enc_bytes) % 2 != 0:
            enc_bytes[len(enc_bytes) // 2] ^= key_bytes[0]

        try:
            result = enc_bytes.decode("utf-8")
        except UnicodeDecodeError:
            result = enc_bytes.hex()
        results.append(result)
    return results
```

##### 总结

远控木马是一种恶意软件，它能让攻击者在用户不知情的情况下远程控制用户的电脑，从而窃取隐私、控制摄像头和麦克风、篡改系统设置、盗取数据，甚至发动网络攻击或部署勒索软件，危害极大。而钓鱼邮件是传播远控木马的常见手段，攻击者通过伪装成合法邮件诱骗用户点击恶意链接或下载附件。

要防范钓鱼邮件，大家要提高安全意识，不轻易点击不明链接或下载附件，仔细检查邮件来源，使用安全软件进行防护，启用双因素认证，验证链接和附件的安全性，及时报告可疑邮件，并定期备份重要数据。只有这样，才能有效降低被远控木马攻击的风险，保护好自己的信息安全。
