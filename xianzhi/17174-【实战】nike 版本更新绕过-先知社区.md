完整代码：

```
function hook_navigate() {
    Java.perform(function () {

        let MainActivity = Java.use("com.nike.mynike.ui.MainActivity");
        MainActivity["setupCurrentFragment"].implementation = function (isTabEnabled, fragmentToShow) {
            console.log(`MainActivity.setupCurrentFragment is called: isTabEnabled=${isTabEnabled}, fragmentToShow=${fragmentToShow}`);
            this["setupCurrentFragment"](true, fragmentToShow);
  
        
        };

        let NikeConfigurationDataImpl = Java.use("com.nike.mynike.configuration.NikeConfigurationDataImpl");
        NikeConfigurationDataImpl["isAppVersionLowerThanMin"].implementation = function (currentVersion, currentAosSdkVersionCode) {
            console.log(`NikeConfigurationDataImpl.isAppVersionLowerThanMin is called: currentVersion=${currentVersion}, currentAosSdkVersionCode=${currentAosSdkVersionCode}`);
            let result = this["isAppVersionLowerThanMin"](currentVersion, currentAosSdkVersionCode);
            console.log(`NikeConfigurationDataImpl.isAppVersionLowerThanMin result=${result}`);
            return false;
        };


    });
}

setImmediate(hook_navigate);
```
