from PIL import Image
import numpy as np


def image_to_hex(image_path, output_path):
    # 打开图像并转换为二值图像
    img = Image.open(image_path).convert('1')

    # 获取图像尺寸
    width, height = img.size
    print(f"图像尺寸: {width}x{height}")

    # 获取像素数据
    pixel_data = np.array(img)

    # 创建一个二进制字符串，黑色(0)为1，白色(255)为0
    binary_string = ''
    for row in pixel_data:
        for pixel in row:
            # 在二值模式下，PIL将白色表示为255，黑色表示为0
            # 但要求是黑色为1，白色为0，所以需要反转
            binary_string += '1' if pixel == 0 else '0'

    # 如果二进制字符串长度不是4的倍数，添加补位0
    if len(binary_string) % 4 != 0:
        padding = 4 - (len(binary_string) % 4)
        binary_string += '0' * padding

    # 将二进制字符串转换为十六进制
    hex_string = ''
    for i in range(0, len(binary_string), 4):
        chunk = binary_string[i:i + 4]
        hex_value = hex(int(chunk, 2))[2:]  # 移除'0x'前缀
        hex_string += hex_value

    # 写入文件
    with open(output_path, 'w') as f:
        f.write(hex_string)

    print(f"已将图像转换为十六进制并保存到 {output_path}")


# 使用示例
if __name__ == "__main__":
    image_path = "output.png"  # 请替换为实际的图像文件路径
    output_path = "output.txt"  # 输出文件路径
    image_to_hex(image_path, output_path)

    if len(data.shape) > 1:
        data = data[:, 0]

    window_size = int(sample_rate * 0.01)
    binary_data = []  # 存储提取的二进制值
    frequencies = []  # 用于记录每个窗口的主频率

    # 对每个窗口进行处理
    for i in range(0, len(data), window_size):
        window = data[i:i + window_size]
        # 跳过不完整的窗口
        if len(window) < window_size:
            break

        # 执行FFT变换
        fft_result = np.fft.fft(window)
        freqs = np.fft.fftfreq(len(window), d=1 / sample_rate)

        # 计算幅度谱
        magnitude = np.abs(fft_result)

        # 找出主频率（只分析0-1000Hz的正频率部分）
        positive_freqs_indices = np.where((freqs > 0) & (freqs < 1000))[0]
        max_index = positive_freqs_indices[np.argmax(magnitude[positive_freqs_indices])]
        dominant_freq = freqs[max_index]

        frequencies.append(dominant_freq)

        # 根据主频率判断二进制值：高于阈值为1，低于阈值为0
        binary_value = 1 if dominant_freq >= threshold_freq else 0
        binary_data.append(binary_value)

    # 将二进制数据转换为字符串
