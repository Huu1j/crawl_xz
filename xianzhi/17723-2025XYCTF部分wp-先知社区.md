        v5 *= 2LL;
      else
        v5 = (2 * v5) ^ 0x42F0E1EBA9EA3693LL;
    }
  }
  return ~v5;
}

from pwn import *

# ---------- 状态恢复辅助函数 ----------
def undo_right(y, shift):
    """逆转： y = x ^ (x >> shift)"""
    result = 0
    for i in range(32):
        bit = (y >> (31 - i)) & 1
        if (31 - i + shift) < 32:
            bit ^= (result >> (31 - i + shift)) & 1
        result |= (bit << (31 - i))
    return result


def undo_left(y, shift, mask):
    """逆转： y = x ^ ((x << shift) & mask)"""
    result = 0
    for i in range(32):
        bit = (y >> i) & 1
        if i - shift >= 0:
            bit ^= ((result >> (i - shift)) & 1) & ((mask >> i) & 1)
        result |= (bit << i)
    return result


def untemper(y):
    """
    逆转 MT19937 tempering 过程：
        y ^= (y >> 11)
        y ^= ((y << 7) & 0x9d2c5680)
        y ^= ((y << 15) & 0xefc60000)
        y ^= (y >> 18)
    """
    y = undo_right(y, 18)
    y = undo_left(y, 15, 0xefc60000)
    y = undo_left(y, 7, 0x9d2c5680)
    y = undo_right(y, 11)
    return y


def recover_state(outputs):
    """
    根据 624 个输出恢复内部状态
    :param outputs: 32 位随机数列表，长度>=624
    :return: 内部状态列表（624 个数）
    """
    assert len(outputs) >= 624, "至少需要 624 个输出！"
    return [untemper(x) for x in outputs[:624]]


def set_local_random_state(recovered_state):
    """
    将恢复的状态设置到 Python 的 random 模块中
    内部状态格式：(version, (state tuple, index), gauss)
    """
    mt_state = (3, tuple(recovered_state + [624]), None)
    random.setstate(mt_state)


def get_predicted_answer():
    """
    预测后续生成的两个大随机数，并计算正确答案
    """
    rand1 = random.getrandbits(11000)
    rand2 = random.getrandbits(10000)
    return rand1 // rand2


# ---------- 与服务器交互部分 ----------
def parse_output(line):
    """
    解析选项 [1] 返回的输出，期望格式类似：
    "nominator//1 = <number>"
    """
    try:
        return int(line.split('=')[-1].strip())
    except Exception as e:

        return None

        p.sendline(b'1')  # 发送除数 1，确保输出为随机数本身

        num = parse_output(line)
        if num is None:

            return
        outputs.append(num)
        log.info("收集 {}/624 输出: {}".format(i + 1, num))

    recovered_state = recover_state(outputs)

    # 如果有更多输出，可以继续交互
    p.interactive()


if __name__ == '__main__':
    main()

### Ret2libc' Revenge

**程序有个无限制的溢出,因此我们就要想办法去寻找有没有其他的gadget可用,因为是无限的溢出，所以我们可以随便跳回来，难点主要是泄露libc，有了libc，就有了各种gadget和函数**

**泄露****libc****肯定是用****puts****函数，也就是说，我们要想办法让****rdi****的值为一个地址，这个地址中存放的是一个****libc****相关的地址**

**发现了这个两个****gadget**

**0x0000000000401180 : mov rdi, rsi ; ret**

```
# .text:00000000004010EB                add 
# .text:00000000004010EF                retn   rsi, [rbp+20h]
```

**只需要让****rbp+0x20****的位置存放的是****5****，让****rsi****加****5****，然后再把****rsi****赋值给****rdi****就行了**

![](images/20250411172507-db82b382-16b6-1.png)

**DT\_STRTAB****：字符串表，****d\_ptr****是字符串表地址**

**d\_tag位置刚好为5**

**那我们就能泄露libc了**

```
0xebc88 execve("/bin/sh", rsi, rdx) constraints: address rbp-0x78 is writable [rsi] == NULL || rsi == NULL || rsi is a valid argv [rdx] == NULL || rdx == NULL || rdx is a valid envp
```
